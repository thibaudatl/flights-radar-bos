function acft(cs, icao) {

  const pool = FLEET[(cs||'').slice(0,3).toUpperCase()] || GEN;

  const seed = parseInt((icao||'0000').slice(-4),16) || 1;

  const tot = pool.reduce((s,e)=>s+e.w,0);

  let p = seed % tot;

  for (const e of pool){p-=e.w;if(p<0)return e;}

  return pool[0];

}

function inferDest(heading) {

  if (heading==null) return null;

  const h=((heading%360)+360)%360;

  if(h>195&&h<255) return {city:'Los Angeles',iata:'LAX'};

  if(h>245&&h<278) return {city:'San Francisco',iata:'SFO'};

  if(h>170&&h<215) return {city:'Miami',iata:'MIA'};

  if(h>258&&h<295) return {city:'Denver',iata:'DEN'};

  if(h>293&&h<335) return {city:'Seattle',iata:'SEA'};

  if(h>28 &&h<72)  return {city:'London',iata:'LHR'};

  if(h>55 &&h<95)  return {city:'Paris',iata:'CDG'};

  if(h>198&&h<232) return {city:'Charlotte',iata:'CLT'};

  if(h>262&&h<302) return {city:'Chicago',iata:'ORD'};

  if(h>315&&h<348) return {city:'Montreal',iata:'YUL'};

// ─── CLASSIFY ─────────────────────────────────────────────────────

function classify(f){

  if(f.onGround) return 'ground';

  const a=f.alt||0,v=f.vertRate||0;

  if(v<-1&&a<8000)  return 'approach';

  if(v>1 &&a<15000) return 'climb';

  if(a>20000)        return 'cruise';

  return 'enroute';

}

function isDep(f){ const t=classify(f); return t==='climb'||t==='ground'; }

function isArr(f){ const t=classify(f); return t==='approach'||(t==='enroute'&&(f.alt||0)<5000); }

function dist(f) { const d=(f.lat||airport.lat)-airport.lat,e=(f.lon||airport.lon)-airport.lon; return Math.sqrt(d*d+e*e)*111; }


function isImmDep(f){ const d=dist(f); return (f.onGround&&d<3)||(classify(f)==='climb'&&(f.alt||0)<2000&&d<5); }

function isImmArr(f){ return classify(f)==='approach'&&(f.alt||0)<1500&&dist(f)<5; }


function pill(f, isDeparture){

  if(isDeparture&&isImmDep(f)) return ['DEPARTING','s-imm-dep'];

  if(!isDeparture&&isImmArr(f)) return ['LANDING','s-imm-arr'];

  const m={approach:['APPROACH','s-approach'],climb:['CLIMBING','s-climb'],cruise:['ENROUTE','s-cruise'],enroute:['ENROUTE','s-enroute'],ground:['ON GROUND','s-ground']};

  return m[classify(f)]||['——','s-ground'];

}


// ─── TIME: never show past times ──────────────────────────────────

function displayTime(f, isDeparture){

  const now=new Date();

  if(isDeparture){

    // Time of departure = now minus how long it's been climbing

    const minsAgo = f.alt ? Math.round(f.alt/1800) : 0;

    const est = new Date(now.getTime()-minsAgo*60000);

    // Clamp: never show more than 1 min in the past

    const clamped = new Date(Math.max(est.getTime(), now.getTime()-60000));

    return fmt(clamped);

  } else {

    // Arrival = always future: now + descent time

    const minsOut = f.alt ? Math.round(f.alt/1400) : 4;

    return fmt(new Date(now.getTime()+Math.max(minsOut,1)*60000));

  }

}

function fmt(d){ return d.toLocaleTimeString('en-US',{hour12:false,hour:'2-digit',minute:'2-digit'}); }


// ─── DEST ─────────────────────────────────────────────────────────

function destInfo(f, isDeparture){

  const cs=f.callsign.trim(), r=routeCache[cs];

  if(r){

    if(isDeparture&&(r.destCity||r.destIata)) return {city:r.destCity||city(r.destIata),iata:r.destIata};

    if(!isDeparture&&(r.origCity||r.origIata)) return {city:r.origCity||city(r.origIata),iata:r.origIata};

  }

  if(isDeparture){ const inf=inferDest(f.heading); if(inf) return inf; }

  return {city:f.country||'——',iata:null};

}

// ─── IMMINENT ALERT ───────────────────────────────────────────────

// ─── ALERT QUEUE SYSTEM ───────────────────────────────────────────

// How it works:

//   1. On each data refresh, ALL imminent flights are collected and sorted

//      by proximity (closest to airport first). Arrivals beat departures

//      at equal distance since a landing plane is more immediately visible.

//   2. Any flight already dismissed this session is skipped.

//   3. If an alert is currently showing, new candidates are queued —

//      they will display automatically after the current one clears.

//   4. Each alert shows for 10s (or until manually dismissed), then the

//      next queued flight fires immediately.

//   5. If multiple flights are simultaneously imminent (e.g. two planes

//      on approach at the same time), each gets its own alert in sequence.


let alertQueue = [];   // pending {f, isDeparture} entries

let alertActive = false;

let autoInterval = null;

let autoCountdown = 10;


function checkAlerts(deps, arrs) {

  // Build full candidate list sorted: arrivals first (more dramatic),

  // then by distance to airport (closest first)

  const cands = [

    ...arrs.filter(isImmArr).map(f=>({f, isDeparture:false, d:dist(f)})),

    ...deps.filter(isImmDep).map(f=>({f, isDeparture:true,  d:dist(f)})),

  ]

  .filter(({f}) => !dismissed.has(f.callsign))

  .sort((a,b) => a.d - b.d);


  // Merge into queue: add any not already queued or showing

  const inQueue = new Set(alertQueue.map(e=>e.f.callsign));

  for (const c of cands) {

    if (!inQueue.has(c.f.callsign)) alertQueue.push(c);

  }


  // If nothing is showing right now, kick off the queue

  if (!alertActive && alertQueue.length > 0) showNextAlert();

}


function showNextAlert() {

  if (alertQueue.length === 0) return;

  const {f, isDeparture} = alertQueue.shift();


  // Skip if dismissed while waiting in queue

  if (dismissed.has(f.callsign)) { showNextAlert(); return; }


  alertActive = true;

  const ov = document.getElementById('immOverlay');

  ov.className = 'imm-overlay show ' + (isDeparture ? 'is-dep' : 'is-arr');


  const d  = destInfo(f, isDeparture);

  const al = airline(f.callsign);

  const a  = acft(f.callsign, f.icao);

  const dst = d.city ? (d.city + (d.iata ? ' · ' + d.iata : '')) : '——';

  const queuedMore = alertQueue.filter(e => !dismissed.has(e.f.callsign)).length;


  document.getElementById('immBadge').textContent    = isDeparture ? '✈  NOW DEPARTING' : '✈  NOW LANDING';

  document.getElementById('immCallsign').textContent = f.callsign;

  document.getElementById('immDest').textContent     = isDeparture ? `→ ${dst}` : `← ${dst}`;

  document.getElementById('immWatermark').style.transform = isDeparture ? 'rotate(-15deg)' : 'rotate(15deg) scaleX(-1)';


  const alStr  = al ? `${al.f} ${al.name}` : f.country || '—';

  const altStr = f.alt   ? `${f.alt.toLocaleString()} ft` : '—';

  const spdStr = f.speed ? `${f.speed} kts` : '—';

  const queueNote = queuedMore > 0 ? `<br><span style="color:#555;font-size:0.7rem">+${queuedMore} MORE IN QUEUE</span>` : '';

  document.getElementById('immMeta').innerHTML =

    `<em>${alStr}</em><br>${a.t} · ${a.s}<br>ALT ${altStr}  ·  SPD ${spdStr}${queueNote}`;


  // Countdown — 10s per flight

  autoCountdown = 10;

  document.getElementById('immAuto').textContent = `AUTO-DISMISS IN ${autoCountdown}s`;

  clearInterval(autoInterval);

  autoInterval = setInterval(() => {

    autoCountdown--;

    document.getElementById('immAuto').textContent = `AUTO-DISMISS IN ${autoCountdown}s`;

    if (autoCountdown <= 0) { clearInterval(autoInterval); dismissAlert(f.callsign); }

  }, 1000);

}


function dismissAlert(cs) {

  alertActive = false;

  clearInterval(autoInterval);

  document.getElementById('immOverlay').className = 'imm-overlay';

  if (cs) dismissed.add(cs);

  // Fire next in queue after a short breath

  setTimeout(() => { if (!alertActive && alertQueue.length > 0) showNextAlert(); }, 600);

}

document.addEventListener('keydown', e => { if (e.key === 'Escape') dismissAlert(); });
